SELECT 'DROP TABLE ' || object_name || ' CASCADE CONSTRAINTS;'
FROM   user_objects
WHERE  object_type = 'TABLE';

DROP TABLE PRODREPLY CASCADE CONSTRAINTS;
DROP TABLE QNAREPLY CASCADE CONSTRAINTS;
DROP TABLE MEMBERS CASCADE CONSTRAINTS;
DROP TABLE PAYMENTS CASCADE CONSTRAINTS;
DROP TABLE CARTS CASCADE CONSTRAINTS;
DROP TABLE CALANDER CASCADE CONSTRAINTS;
DROP TABLE QNA CASCADE CONSTRAINTS;
DROP TABLE CATEGORY CASCADE CONSTRAINTS;
DROP TABLE PRODUCTS CASCADE CONSTRAINTS;

CREATE TABLE qnareply (
	qna_revno	NUMBER(4) NOT NULL,
	qna_content	varchar2(1000) NULL,
	user_id	varchar2(20) NULL,
	qna_revDate	DATE NULL,
	qna_no	number NULL
);


CREATE TABLE products (
	prod_no	number NOT NULL,
	prod_name	varchar2(1000) NULL,
	price	number NULL,
	content	varchar2(1000) NULL,
	create_date	DATE DEFAULT sysdate,
	image	varchar2(2000) NULL,
	pdf	VARCHAR2(2000) NULL,
	count	NUMBER DEFAULT 0,
	user_id	VARCHAR2(200) NULL,
    	page  number NULL,
	category_no	varchar2(10)	NULL
);



CREATE TABLE prodReply (
	rev_no	number	NOT NULL,
	star_point	number	NULL,
	rev_title	VARCHAR2(2000)	NULL,
	rev_create	DATE	NULL,
	rev_content	VARCHAR2(1000)	NULL,
	user_id	VARCHAR2(200)	NOT NULL,
	pay_no	number	NOT NULL
);


CREATE TABLE members (
	user_id	VARCHAR2(200) NOT NULL,
	user_name	VARCHAR2(200) NULL,
	phone	VARCHAR2(200) NULL,
	email	VARCHAR2(1000) NULL,
	birth	DATE NULL,
	password	VARCHAR2(200) NULL,
	user_type	VARCHAR2(100) NULL,
	out_type	VARCHAR2(100)	DEFAULT 'no'	NULL
);

CREATE TABLE payments (
	pay_no	number	NOT NULL,
	price	number	NULL,
	prod_no	number	NULL,
	pay_date	DATE	DEFAULT sysdate,
	user_id	varchar2(200)	NULL,
	pay_type	varchar2(100)	NULL,
	down_cnt	NUMBER	DEFAULT 0 
);




CREATE TABLE carts (
	cart_no	number NOT NULL,
	user_id	VARCHAR2(200) NULL,
	prod_no	number NULL
);

CREATE TABLE calander(
   title VARCHAR2(1000),
   start_date  VARCHAR2(600),
   end_date VARCHAR2 (600)
);

CREATE TABLE qna (
	qna_no	number NOT NULL,
	qna_title	varchar2(1000) NULL,
	qna_content	varchar2(1000) NULL,
	qna_create	DATE NULL,
	qna_cnt	number NULL,
	user_id	varchar2(200) NULL,
	category_no	varchar2(100)	NULL
);



CREATE TABLE category (
	category_no	varchar2(100) NOT NULL,
	category_name	VARCHAR2(100) NULL
);
ALTER TABLE qnareply ADD CONSTRAINT PK_QNAREPLY PRIMARY KEY (
	qna_revno
);

ALTER TABLE products ADD CONSTRAINT PK_PRODUCTS PRIMARY KEY (
	prod_no
);

ALTER TABLE prodReply ADD CONSTRAINT PK_PRODREPLY PRIMARY KEY (
	rev_no
);


ALTER TABLE members ADD CONSTRAINT PK_MEMBERS PRIMARY KEY (
	user_id
);

ALTER TABLE payments ADD CONSTRAINT PK_PAYMENTS PRIMARY KEY (
	pay_no
);

ALTER TABLE carts ADD CONSTRAINT PK_CARTS PRIMARY KEY (
	cart_no
);

ALTER TABLE qna ADD CONSTRAINT PK_QNA PRIMARY KEY (
	qna_no
);

ALTER TABLE category ADD CONSTRAINT PK_CATEGORY PRIMARY KEY (
	category_no
);
